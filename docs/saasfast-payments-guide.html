CREATE TABLE store_transactions ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE, -- المعرفات الخارجية myfatoorah_invoice_id VARCHAR(100), -- رقم الفاتورة في MyFatoorah myfatoorah_payment_id VARCHAR(100), -- رقم الدفعة store_order_id UUID, -- رابط بطلب المتجر (إن وُجد) payment_link_id UUID, -- رابط برابط الدفع (إن وُجد) external_reference VARCHAR(200), -- مرجع خارجي -- المبالغ gross_amount NUMERIC(12,2) NOT NULL, -- المبلغ الإجمالي currency VARCHAR(3) DEFAULT 'SAR', -- الرسوم (تُحسَب تلقائياً) gateway_fee_rate NUMERIC(5,4) DEFAULT 0.015, -- 1.5% افتراضي gateway_fee_amount NUMERIC(12,2) GENERATED ALWAYS AS (ROUND(gross_amount * gateway_fee_rate, 2)) STORED, platform_fee_rate NUMERIC(5,4) DEFAULT 0.01, -- 1% عمولة المنصة platform_fee_amount NUMERIC(12,2) GENERATED ALWAYS AS (ROUND(gross_amount * platform_fee_rate, 2)) STORED, net_amount NUMERIC(12,2) GENERATED ALWAYS AS (gross_amount - ROUND(gross_amount * gateway_fee_rate, 2) - ROUND(gross_amount * platform_fee_rate, 2)) STORED, -- طريقة الدفع payment_method VARCHAR(50), -- mada | visa | mastercard | apple_pay | stcpay payment_network VARCHAR(50), -- Visa | Mastercard | Mada card_last4 VARCHAR(4), card_brand VARCHAR(20), -- معلومات العميل customer_name VARCHAR(200), customer_email VARCHAR(255), customer_phone VARCHAR(20), -- الحالة status VARCHAR(30) NOT NULL DEFAULT 'pending', -- pending | success | failed | refunded | partial_refund | chargeback -- الاسترداد refunded_amount NUMERIC(12,2) DEFAULT 0, refund_reason TEXT, refunded_at TIMESTAMPTZ, -- الإيداع settlement_id UUID, -- رابط بعملية التسوية settled_at TIMESTAMPTZ, -- البيانات الخام من MyFatoorah raw_payload JSONB, created_at TIMESTAMPTZ DEFAULT NOW(), updated_at TIMESTAMPTZ DEFAULT NOW() ); -- Indexes للأداء CREATE INDEX idx_store_tx_tenant ON store_transactions(tenant_id, created_at DESC); CREATE INDEX idx_store_tx_status ON store_transactions(status); CREATE INDEX idx_store_tx_settlement ON store_transactions(settlement_id); CREATE UNIQUE INDEX idx_store_tx_mf_invoice ON store_transactions(myfatoorah_invoice_id) WHERE myfatoorah_invoice_id IS NOT NULL; -- RLS ALTER TABLE store_transactions ENABLE ROW LEVEL SECURITY; CREATE POLICY tx_tenant_isolation ON store_transactions FOR ALL USING (tenant_id = current_setting('app.current_tenant_id')::UUID);

CREATE TABLE merchant_balances ( id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), tenant_id UUID UNIQUE NOT NULL REFERENCES tenants(id) ON DELETE CASCADE, -- الأرصدة available_balance NUMERIC(12,2) DEFAULT 0, ← جاهز للتحويل pending_balance NUMERIC(12,2) DEFAULT 0, ← قيد المعالجة (T+1) reserved_balance NUMERIC(12,2) DEFAULT 0, ← محجوز للردود/النزاعات total_earned NUMERIC(12,2) DEFAULT 0, ← إجمالي ما كسبه total_withdrawn NUMERIC(12,2) DEFAULT 0, ← إجمالي ما حوّله -- الإحصائيات total_transactions INTEGER DEFAULT 0, successful_transactions INTEGER DEFAULT 0, total_refunded NUMERIC(12,2) DEFAULT 0, currency VARCHAR(3) DEFAULT 'SAR', last_updated TIMESTAMPTZ DEFAULT NOW() ); -- دالة تحديث الرصيد تلقائياً عند كل عملية ناجحة CREATE OR REPLACE FUNCTION update_merchant_balance_on_transaction() RETURNS TRIGGER AS $$ BEGIN IF NEW.status = 'success' AND (OLD.status IS NULL OR OLD.status != 'success') THEN INSERT INTO merchant_balances(tenant_id, pending_balance, total_earned, total_transactions, successful_transactions) VALUES(NEW.tenant_id, NEW.net_amount, NEW.net_amount, 1, 1) ON CONFLICT(tenant_id) DO UPDATE SET pending_balance = merchant_balances.pending_balance + NEW.net_amount, total_earned = merchant_balances.total_earned + NEW.net_amount, total_transactions = merchant_balances.total_transactions + 1, successful_transactions = merchant_balances.successful_transactions + 1, last_updated = NOW(); END IF; RETURN NEW; END; $$ LANGUAGE plpgsql; CREATE TRIGGER trg_update_balance AFTER INSERT OR UPDATE ON store_transactions FOR EACH ROW EXECUTE FUNCTION update_merchant_balance_on_transaction();

نظام التسويات

من طلب التحويل إلى الإيداع البنكي

دورة حياة كاملة: التاجر يطلب → الإدارة تراجع وتحوّل → التاجر يستلم إشعاراً.

1. التاجر يطلب تحويل

// POST /api/payments/withdrawal-request export async function POST(req: Request) { const { amount, bank_account_id } = await req.json() const tenantId = session.user.tenant_id // تحقق من الرصيد المتاح const balance = await getMerchantBalance(tenantId) if (amount > balance.available_balance) throw Error('الرصيد غير كافٍ') // أنشئ طلب تسوية const settlement = await supabase .from('settlements').insert({ tenant_id: tenantId, bank_account_id, settlement_number: generateSettlementNumber(), gross_amount: amount, gateway_fees: calcFees(amount, 'gateway'), platform_fees: calcFees(amount, 'platform'), net_amount: amount - fees, period_start: lastSettledAt, period_end: new Date(), status: 'pending' }).select().single() // حجز المبلغ من الرصيد المتاح await reserveBalance(tenantId, amount) return { settlement } }

2. الإدارة تنفذ التحويل

// POST /api/admin/payments/settle // (super_admin فقط) export async function POST(req: Request) { const { settlement_id, bank_reference, notes } = await req.json() // 1. تحديث حالة التسوية await supabase.from('settlements').update({ status: 'transferred', bank_reference, transferred_at: new Date(), approved_by: adminId, notes }).eq('id', settlement_id) // 2. تحديث رصيد التاجر await supabase.from('merchant_balances').update({ available_balance: balance - netAmount, reserved_balance: reserved - netAmount, total_withdrawn: withdrawn + netAmount, }).eq('tenant_id', tenantId) // 3. ربط العمليات بالتسوية await supabase.from('store_transactions').update({ settlement_id, settled_at: new Date() }).eq('tenant_id', tenantId).is('settlement_id', null) // 4. أرسل إيميل للتاجر await sendSettlementEmail(tenantId, settlement) // 5. سجّل في audit_logs await logAudit({ action: 'SETTLEMENT_TRANSFERRED', ... }) }

رسوم البوابة والعمولة — قابلة للتخصيص لكل خطة

-- في جدول plans (موجود) — أضف الحقول: ALTER TABLE plans ADD COLUMN IF NOT EXISTS gateway_fee_rate NUMERIC(5,4) DEFAULT 0.015; -- 1.5% افتراضي ALTER TABLE plans ADD COLUMN IF NOT EXISTS platform_fee_rate NUMERIC(5,4) DEFAULT 0.01; -- 1.0% افتراضي -- مثال: الخطة الاحترافية تدفع أقل -- Basic: gateway=1.5%, platform=1.5% -- Professional: gateway=1.5%, platform=1.0% -- Enterprise: gateway=1.5%, platform=0.5% -- دالة جلب الرسوم حسب الخطة: CREATE OR REPLACE FUNCTION get_tenant_fee_rates(p_tenant_id UUID) RETURNS TABLE(gateway_rate NUMERIC, platform_rate NUMERIC) AS $$ SELECT p.gateway_fee_rate, p.platform_fee_rate FROM tenants t JOIN subscriptions s ON s.tenant_id = t.id AND s.status = 'active' JOIN plans p ON p.id = s.plan_id WHERE t.id = p_tenant_id LIMIT 1; $$ LANGUAGE SQL;
