🔑 منطق التحقق من الكوبون

// src/lib/discounts/validate.ts export async function validateCoupon( code: string, tenantId: string, orderData: { amount: number; products: string[] } ) { // 1. تحقق الوجود والنشاط const discount = await supabase .from('discounts') .select('*') .eq('tenant_id', tenantId) .eq('code', code.toUpperCase()) .eq('is_active', true) .single() // 2. تحقق انتهاء الصلاحية if (discount.ends_at && new Date() > new Date(discount.ends_at)) throw new Error('انتهت صلاحية الكوبون') // 3. تحقق الحد الأقصى للاستخدام if (discount.max_uses && discount.used_count >= discount.max_uses) throw new Error('تجاوز الكوبون الحد الأقصى') // 4. تحقق الحد الأدنى للطلب if (discount.min_order_amount && orderData.amount < discount.min_order_amount) throw new Error(`الحد الأدنى للطلب ${discount.min_order_amount} ر.س`) // 5. احسب قيمة الخصم return { discount, savings: calculateSavings(discount, orderData) } }

منطق كسب النقاط

// src/lib/marketing/loyalty.ts export async function awardPoints(tenantId: string, customerId: string, orderId: string, orderAmount: number) { const program = await getLoyaltyProgram(tenantId) if (!program?.is_active) return const points = Math.floor(orderAmount * program.points_per_sar) // احسب الرصيد الجديد const currentBalance = await getCustomerPoints(tenantId, customerId) const newBalance = currentBalance + points // سجّل المعاملة await supabase.from('loyalty_transactions').insert({ tenant_id: tenantId, customer_id: customerId, type: 'earn', points, balance_after: newBalance, reference_id: orderId }) return { points, newBalance } } // يُستدعى من webhook الدفع الناجح // في: app/api/payments/webhook/route.ts

استرداد النقاط عند الدفع

export async function redeemPoints( tenantId: string, customerId: string, pointsToRedeem: number ) { const program = await getLoyaltyProgram(tenantId) const balance = await getCustomerPoints(tenantId, customerId) if (balance < pointsToRedeem) throw new Error('رصيد النقاط غير كافٍ') // قيمة الخصم بالريال const discount = pointsToRedeem * program.sar_per_point await supabase.from('loyalty_transactions').insert({ type: 'redeem', points: -pointsToRedeem, balance_after: balance - pointsToRedeem, }) return { discount } // يُطبَّق كخصم على الطلب }
